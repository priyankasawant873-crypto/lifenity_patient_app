    <script>

        // App store URLs

        const ANDROID_URL = 'https://play.google.com/store/apps/details?id=com.lifenityhealth.connect';

        const IOS_URL = 'https://apps.apple.com/in/app/lifenity-connect/id6749895597';

       

        function detectDevice() {

            const userAgent = navigator.userAgent.toLowerCase();

            const isAndroid = /android/.test(userAgent);

            const isIOS = /iphone|ipad|ipod/.test(userAgent);

           

            return { isAndroid, isIOS };

        }

        

        function updateUI(device) {

            const loadingElement = document.getElementById('loading');

            const manualLinksElement = document.getElementById('manual-links');

            const androidLinkElement = document.getElementById('android-link');

            const iosLinkElement = document.getElementById('ios-link');

           

            // Set the actual URLs

            androidLinkElement.href = ANDROID_URL;

            iosLinkElement.href = IOS_URL;

           

            if (device.isAndroid) {

                loadingElement.innerHTML = 'Redirecting to Google Play Store...<br><span style="font-size: 0.9em; opacity: 0.8;">Google Play Store वर पाठवत आहे...</span>';

                setTimeout(() => {

                    window.location.href = ANDROID_URL;

                }, 2000);

            } else if (device.isIOS) {

                loadingElement.innerHTML = 'Redirecting to App Store...<br><span style="font-size: 0.9em; opacity: 0.8;">App Store वर पाठवत आहे...</span>';

                setTimeout(() => {

                    window.location.href = IOS_URL;

                }, 2000);

            } else {

                loadingElement.style.display = 'none';

                manualLinksElement.style.display = 'block';

            }

        }

       

        // Run detection when page loads

        window.addEventListener('load', () => {

            const device = detectDevice();

            updateUI(device);

        });

       

        // Fallback: Show manual links after 5 seconds if redirect fails

        setTimeout(() => {

            document.getElementById('manual-links').style.display = 'block';

        }, 5000);

    </script>
